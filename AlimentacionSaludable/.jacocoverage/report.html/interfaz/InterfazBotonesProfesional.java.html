    public void actualizarFotoYNombre(){
        ImageIcon fotoUsuario = usuarioActivo.getFotoPerfil();

        etiquetaFotoProfesional.setIcon(fotoProfesional);
        etiquetaNombreProfesional.setText(usuarioActivo.getNombreUsuario());
    }
